CREATE DATABASE IF NOT EXISTS pedidos_venda
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_general_ci;

USE pedidos_venda;

CREATE TABLE IF NOT EXISTS clientes (
    codigo  INT          NOT NULL AUTO_INCREMENT,
    nome    VARCHAR(100) NOT NULL,
    cidade  VARCHAR(80)  NOT NULL,
    uf      CHAR(2)      NOT NULL,
    CONSTRAINT pk_clientes PRIMARY KEY (codigo)
) ENGINE = InnoDB
  DEFAULT CHARSET = utf8mb4
  COLLATE = utf8mb4_general_ci;

CREATE INDEX idx_clientes_nome ON clientes (nome);
CREATE TABLE IF NOT EXISTS produtos (
    codigo      INT            NOT NULL AUTO_INCREMENT,
    descricao   VARCHAR(100)   NOT NULL,
    preco_venda DECIMAL(10, 2) NOT NULL DEFAULT 0.00,
    CONSTRAINT pk_produtos PRIMARY KEY (codigo)
) ENGINE = InnoDB
  DEFAULT CHARSET = utf8mb4
  COLLATE = utf8mb4_general_ci;

CREATE INDEX idx_produtos_descricao ON produtos (descricao);
CREATE TABLE IF NOT EXISTS pedidos (
    numero_pedido  INT            NOT NULL AUTO_INCREMENT,
    data_emissao   DATE           NOT NULL,
    codigo_cliente INT            NOT NULL,
    valor_total    DECIMAL(10, 2) NOT NULL DEFAULT 0.00,
    CONSTRAINT pk_pedidos          PRIMARY KEY (numero_pedido),
    CONSTRAINT fk_pedidos_clientes FOREIGN KEY (codigo_cliente)
        REFERENCES clientes (codigo)
        ON UPDATE CASCADE
        ON DELETE RESTRICT
) ENGINE = InnoDB
  DEFAULT CHARSET = utf8mb4
  COLLATE = utf8mb4_general_ci;

CREATE INDEX idx_pedidos_cliente      ON pedidos (codigo_cliente);
CREATE INDEX idx_pedidos_data_emissao ON pedidos (data_emissao);

CREATE TABLE IF NOT EXISTS pedido_itens (
    id             INT            NOT NULL AUTO_INCREMENT,
    numero_pedido  INT            NOT NULL,
    codigo_produto INT            NOT NULL,
    quantidade     DECIMAL(10, 3) NOT NULL DEFAULT 0.000,
    vr_unitario    DECIMAL(10, 2) NOT NULL DEFAULT 0.00,
    vr_total       DECIMAL(10, 2) NOT NULL DEFAULT 0.00,
    CONSTRAINT pk_pedido_itens   PRIMARY KEY (id),
    CONSTRAINT fk_itens_pedidos  FOREIGN KEY (numero_pedido)
        REFERENCES pedidos (numero_pedido)
        ON UPDATE CASCADE
        ON DELETE CASCADE,
    CONSTRAINT fk_itens_produtos FOREIGN KEY (codigo_produto)
        REFERENCES produtos (codigo)
        ON UPDATE CASCADE
        ON DELETE RESTRICT
) ENGINE = InnoDB
  DEFAULT CHARSET = utf8mb4
  COLLATE = utf8mb4_general_ci;

CREATE INDEX idx_itens_pedido  ON pedido_itens (numero_pedido);
CREATE INDEX idx_itens_produto ON pedido_itens (codigo_produto);
